simplesim

library(tidyverse)
library(magrittr)
library(broom)
library(broom.mixed)
library(brms)
library(rethinking)
library(bayesplot)
library(marginaleffects)

theme_set(theme_bw())
options(
  mc.cores = parallel::detectCores(),
  brms.backend = "cmdstanr"
)

set.seed(19780503)
n = 10
x = sort(runif(n, min = 0, max = 100))
true_alpha = 5
true_beta = 2
true_sigma = 10

y = true_alpha + true_beta * x + rnorm(n, mean = 0, sd = true_sigma)

(dat = tibble(x = x, y = y))
ggplot(dat, aes(x = x, y = y)) + geom_point() + geom_smooth(method = "lm", se = FALSE)

ff = lm(y ~ x, dat)
tidy(ff, conf.int = TRUE)
predictions(ff) %>% tibble
bf = brm(y ~ x, dat)
mcmc_trace(bf)

tidy(bf)
mcmc_intervals(bf)

rt = ulam(alist(
  y ~ dnorm(mu, sigma),
  mu <- alpha + beta * x,
  alpha ~ dnorm(0, 100),
  beta ~ dnorm(0, 100),
  sigma ~ dhalfnorm(0, 10)), data = dat, chains = 4, cores = 4)
trankplot(rt)

precis(rt)
coeftab_plot(coeftab(rt))
